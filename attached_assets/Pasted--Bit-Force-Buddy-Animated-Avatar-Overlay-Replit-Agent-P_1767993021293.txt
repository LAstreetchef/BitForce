**AVATAR CONTROLLER SCRIPT EXAMPLE:**
```javascript
class BitForceBuddy {
  constructor() {
    this.avatar = document.getElementById('bitforce-buddy');
    this.video = document.querySelector('video');
    this.setupEventListeners();
  }
  
  setupEventListeners() {
    this.video.addEventListener('timeupdate', this.onVideoTimeUpdate.bind(this));
    this.video.addEventListener('play', this.onVideoPlay.bind(this));
    this.video.addEventListener('pause', this.onVideoPause.bind(this));
  }
  
  onVideoTimeUpdate() {
    const time = this.video.currentTime;
    
    // Trigger animations based on time
    if (time > 5 && time < 6) this.animate('wave');
    if (time > 12 && time < 13) this.animate('point', '.problem-text');
    // ... more timed events
  }
  
  animate(action, target) {
    this.avatar.classList.remove('wave', 'point', 'celebrate');
    this.avatar.classList.add(action);
    
    if (target) {
      this.highlightElement(target);
    }
  }
  
  highlightElement(selector) {
    // Add visual highlight to targeted element
  }
}
```
