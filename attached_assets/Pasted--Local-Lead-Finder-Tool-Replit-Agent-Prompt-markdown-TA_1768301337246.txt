// Radius calculation and bounding box
function createGeoFence(center, radiusMiles) {
  const earthRadius = 3959; // miles
  const latR = radiusMiles / earthRadius;
  const lngR = latR / Math.cos(center.lat * Math.PI / 180);
  
  return {
    minLat: center.lat - (latR * 180 / Math.PI),
    maxLat: center.lat + (latR * 180 / Math.PI),
    minLng: center.lng - (lngR * 180 / Math.PI),
    maxLng: center.lng + (lngR * 180 / Math.PI)
  };
}

**LEAD SCORING ALGORITHM:**

```python
class LeadScorer:
    def calculate_score(self, lead_data):
        score = 0
        
        # Business type scoring
        if lead_data['type'] in ['tech', 'consulting', 'digital']: score += 30
        elif lead_data['type'] in ['retail', 'service']: score += 20
        else: score += 10
        
        # Online presence scoring
        if lead_data['has_website']: score += 20
        if lead_data['has_social']: score += 15
        if lead_data['recent_news']: score += 25  # Recently in news
        
        # Size indicators
        if lead_data['employee_count'] > 10: score += 15
        if lead_data['revenue_indicator'] == 'growing': score += 20
        
        # Contact availability
        if lead_data['has_email']: score += 10
        if lead_data['has_phone']: score += 5
        
        return min(100, score)  # Cap at 100
```
