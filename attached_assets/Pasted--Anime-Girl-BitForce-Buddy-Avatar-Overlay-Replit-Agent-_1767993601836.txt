class AnimeBitForceBuddy {
  constructor() {
    this.character = new AnimeCharacter();
    this.expressions = new ExpressionManager();
    this.effects = new ParticleSystem();
    this.sync = new VideoSync();
  }
  
  setExpression(name, intensity = 1) {
    // Change facial expression
    this.expressions.set(name);
    
    // Add appropriate effects
    switch(name) {
      case 'excited':
        this.effects.spawnStars(intensity * 10);
        this.character.bounce();
        break;
      case 'thinking':
        this.effects.spawnThoughtBubbles();
        this.character.tiltHead();
        break;
      // ... more expressions
    }
  }
  
  pointAt(element, duration = 2000) {
    this.character.point();
    this.effects.highlightElement(element, duration);
  }
}
